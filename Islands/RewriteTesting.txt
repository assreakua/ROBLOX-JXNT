--// Get players island position
function getIslandFromPos(pos)
local islands = workspace:WaitForChild("Islands"):GetChildren()
for i = 1, #islands do
local island = islands[i]
if island:FindFirstChild("Root") and math.abs(island.PrimaryPart.Position.X - pos.X) <= 1000 and math.abs(island.PrimaryPart.Position.Z - pos.Z) <= 1000 then
return island
end
end
return nil
end

function findNearSlimes()
local slimes = {}
local region = Region3.new(game.Players.LocalPlayer.Character.HumanoidRootPart.Position - Vector3.new(20, 20, 20), game.Players.LocalPlayer.Character.HumanoidRootPart.Position + Vector3.new(20, 20, 20))
for i, v in pairs(game.Workspace:FindPartsInRegion3(region, game.Players.LocalPlayer.Character, math.huge)) do
if v.Parent:IsA("Model") then
table.insert(slimes, v.Parent)
end
end
return slimes
end

function SlimeThing()
for i, v in pairs(game.Workspace.WildernessIsland.Entities:GetChildren()) do
if v.Name == "slimePink" or v.Name == "slimeBlue" or v.Name == "slime" then
repeat
wait(.00001)
_G.ToSeeIfEnabled = false
critthing = math.random(1,3) == 1 and true or false
spawn(function()
for i,v in pairs(findNearSlimes()) do
game:GetService("ReplicatedStorage").Remotes.Functions["CLIENT_SWING_SWORD"]:InvokeServer({["hitUnit"] = v, ["crit"] = critthing})
end
end)
game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v:WaitForChild("HumanoidRootPart").CFrame * CFrame.new(0,-5,0)
game:GetService("ReplicatedStorage").Remotes.Functions["CLIENT_SWING_SWORD"]:InvokeServer({["hitUnit"] = v, ["crit"] = critthing})
until not v:IsDescendantOf(game) or not slimes or not AlsoDoingBuffalo or not AlsoDoingWizard or not AlsoDoingSkorps
elseif not game.Workspace.WildernessIsland.Entities:FindFirstChild("slimePink") and not game.Workspace.WildernessIsland.Entities:FindFirstChild("slimeBlue") and not game.Workspace.WildernessIsland.Entities:FindFirstChild("slime") then

_G.ToSeeIfEnabled = true
